# Secrets and Credentials Patterns
# Each line is a regex pattern to detect potential secrets
# Lines starting with # are comments

# API Keys - Generic
api[_-]?key\s*[:=]\s*['"][a-zA-Z0-9_-]{20,}['"]
apikey\s*[:=]\s*['"][a-zA-Z0-9_-]{20,}['"]
API_KEY\s*=\s*['"]?[a-zA-Z0-9_-]{20,}['"]?

# Google API Keys
AIza[0-9A-Za-z_-]{35}
GEMINI_API_KEY\s*=\s*['"]?[a-zA-Z0-9_-]{30,}['"]?

# OpenAI
OPENAI_API_KEY\s*=\s*['"]?sk-[a-zA-Z0-9]{40,}['"]?
sk-[a-zA-Z0-9]{40,}

# AWS

AWS_ACCESS_KEY_ID\s*=
AWS_SECRET_ACCESS_KEY\s*=

# Azure

AZURE_CLIENT_SECRET\s*=

# Firebase

# Passwords

# Private Keys

# Tokens
token\s*[:=]\s*['"][a-zA-Z0-9_-]{20,}['"]
bearer\s+[a-zA-Z0-9_-]{20,}
auth[_-]?token\s*[:=]

# Database Connection Strings
